import discord
from discord.ext import commands

bot = commands.Bot(command_prefix = '-', intents = discord.Intents.all())

@bot.command('start')
async def start(ctx):
    await ctx.send(f'Привет! Это Это-бот,Мы комьюнити, которое топит за сохранность нашей планеты, здесь ты найдешь информацию о том, как правильно сортировать или перерабатывать тот или иной материал! Сделай вклад в планету, оставив на ней свой бережный след')
    await ctx.send('напиши в чате дерево/пластик/батарейка, чтобы бот дал тебе информацию как их утелизировать')

@bot.command('дерево')
async def wood(ctx):
    await ctx.send('Используют несколько способов утилизации древесины — сжигание, быстрый пиролиз и газификацию. Сжигание древесных отходов производится в специальных печах, в которых образуется тепло для дальнейшего обогрева воды, помещений и так далее.')

@bot.command('пластик')
async def plastic(ctx):
    await ctx.send('Промойте бутылочку или другую пластиковую емкость. Просушите её. Удалите этикетки. Снимите крышки и колечки от них. Сомните бутылку, чтобы она занимала меньше места. Не выбрасывайте пластик в обычную урну - заберите бутылочку домой. Накопленный пластик отнесите в точку сбора вторсырья.')

@bot.command('батарейки')
async def bateries(ctx):
    await ctx.send('Утилизация батареек должна происходить согласно санитарным нормам, то есть в специализированный контейнер. На каждом таком изделии есть обозначение, что его нельзя выкидывать вместе с обычными бытовыми отходами.')
